const DAY_MASTERS = ["Солнце", "Луна", "Марс", "Меркурий", "Юпитер", "Венера", "Сатурн"];
const MS_PER_DAY = 86400000;
const LUNAR_CYCLE = 29.530588853 * MS_PER_DAY;
const NEW_MOON_REF = new Date('2026-04-17T11:52:00Z').getTime();

function getLunarData(date) {
let p = ((date.getTime() - NEW_MOON_REF) % LUNAR_CYCLE) / LUNAR_CYCLE;
if (p < 0) p += 1;
const phase = Math.min(0.9999, Math.max(0, p));
return { phase, age: Math.floor(phase * 29.530588853) };
}

function getPhaseName(phase) {
if (phase < 0.02 || phase > 0.98) return "Новолуние";
if (phase < 0.48) return "Растущая";
if (phase < 0.52) return "Полнолуние";
return "Убывающая";
}

function getAstroPost() {
const moscowTime = new Date().toLocaleString("en-US", { timeZone: "Europe/Moscow" });
const moscowDate = new Date(moscowTime);

    const day = moscowDate.getDate();
    const month = moscowDate.getMonth();
    const year = moscowDate.getFullYear();
    const monthNames = ["января", "февраля", "марта", "апреля", "мая", "июня", "июля", "августа", "сентября", "октября", "ноября", "декабря"];
    const formattedDate = `${day} ${monthNames[month]}`;

    const dayIndex = moscowDate.getDay();
    const planet = DAY_MASTERS[dayIndex];
    const lunar = getLunarData(moscowDate);
    const phase = getPhaseName(lunar.phase);

    // Выбор аффирмации
    const variants = COSMIC_AFFIRMATIONS[planet]?.[phase];
    const seed = (day + month + year) % (variants ? variants.length : 1);
    const affirmation = variants ? variants[seed] : "Я начинаю день с уверенностью.";

    // Выбор пожелания дня
    const dailyVariants = DAILY_WISHES[dayIndex];
    const dailyWish = dailyVariants[Math.floor(Math.random() * dailyVariants.length)];

    // Чистая текстовая верстка без лишних символов
    const postText = [
        `${formattedDate}`,
        `Ваше доброе послание:`,
        '',
        `**${affirmation.replace(/\.$/, '')}**`,
        '',
        `${dailyWish}`,
        '',
        `— — — — — — — — — —`,
        `[Гороскоп • NOVA — поделись и подпишись](https://max.ru/id760407796785_biz)`
    ].join('\n');

    return { postText };
}
